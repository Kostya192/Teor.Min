1) –°—Ä–µ–¥–Ω—è—è —Å–∫–æ—Ä–æ—Å—Ç—å

$$\boxed{\vec{V} = \frac {\vec S} t \text{ [–º/—Å]}} $$ $$\vec{V} - \text{—Å—Ä–µ–¥–Ω—è—è —Å–∫–æ—Ä–æ—Å—Ç—å —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å)} $$ $$\vec S - \text{–ø–µ—Ä–µ–º–µ—â–µ–Ω–∏–µ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º)} $$ $$t - \text{–≤—Ä–µ–º—è –∑–∞ –∫–æ—Ç–æ—Ä–æ–µ —Å–¥–µ–ª–∞–Ω–æ –ø–µ—Ä–µ–º–µ—â–µ–Ω–∏–µ¬†(—Å)} $$

2) –°—Ä–µ–¥–Ω–µ–ø—É—Ç–µ–≤–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å

$$\boxed{{V} = \frac {L} t \text{ [–º/—Å]}}$$ $$V - \text{—Å—Ä–µ–¥–Ω–µ–ø—É—Ç–µ–≤–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å (–º/—Å)}$$ $$L - \text{–ø—É—Ç—å (–º)}$$ $$t¬†-¬†\text{–≤—Ä–µ–º—è¬†(—Å)}$$

–†–∞–≤–Ω–æ—É—Å–∫–æ—Ä–µ–Ω–Ω–æ–µ –¥–≤–∏–∂–µ–Ω–∏–µ

3) –ó–∞–≤–∏—Å–∏–º–æ—Å—Ç—å –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç –æ—Ç –≤—Ä–µ–º–µ–Ω–∏ –ø—Ä–∏ —Ä–∞–≤–Ω–æ—É—Å–∫–æ—Ä–µ–Ω–Ω–æ–º –¥–≤–∏–∂–µ–Ω–∏–∏

$$\boxed{X(t) = X_{0} + V_{0_{x}}\cdot t + {{a_{x} \cdot t^2 \over 2}} \text { [–º]}}$$ $$X_{0} - \text{–Ω–∞—á–∞–ª—å–Ω–∞—è –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞ —Ç–µ–ª–∞ (–º)}\\$$ $$V_{0_{x}} - \text{–Ω–∞—á–∞–ª—å–Ω–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å)}\\$$ $$a_{x} - \text{—É—Å–∫–æ—Ä–µ–Ω–∏–µ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å¬≤)} \\$$ $$t - \text{—Ä–∞—Å—Å–º–∞—Ç—Ä–∏–≤–∞–µ–º—ã–π –ø—Ä–æ–º–µ–∂—É—Ç–æ–∫ –≤—Ä–µ–º–µ–Ω–∏ (—Å) } \\$$ $$X - \text{–∫–æ–Ω–µ—á–Ω–∞—è –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞ —Ç–µ–ª–∞ (–º)} $$

4) –ü–µ—Ä–µ–º–µ—â–µ–Ω–∏–µ –ø—Ä–∏ —Ä–∞–≤–Ω–æ—Å—É–∫–æ—Ä–µ–Ω–Ω–æ–º –¥–≤–∏–∂–µ–Ω–∏–∏ (—Ç—Ä–∏ —Ñ–æ—Ä–º—É–ª—ã)

$$\boxed{ S_{x} = {V_{–ö_{x}}^2 - V_{0_{x}}^2 \over 2a_{x}} \text { [–º]}}$$ $$S_{x} - \text{–ø–µ—Ä–µ–º–µ—â–µ–Ω–∏–µ(–º)} $$ $$V_{–ö_{x}} - \text{–∫–æ–Ω–µ—á–Ω–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å)} \\ $$ $$V_{0_{x}} - \text{–Ω–∞—á–∞–ª—å–Ω–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è(–º/—Å)}\\ $$ $$a_{x} - \text{—É—Å–∫–æ—Ä–µ–Ω–∏–µ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å¬≤)} \\ $$

$$ \boxed{S_{x} = {V_{–ö_{x}} + V_{0_{x}} \over 2} \cdot t \text{ [–º]}}$$ $$S_{x} - \text{–ø–µ—Ä–µ–º–µ—â–µ–Ω–∏–µ(–º)} $$ $$V_{0_{x}} - \text{–Ω–∞—á–∞–ª—å–Ω–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è(–º/—Å)} $$ $$V_{–ö_{x}} \text{ - –∫–æ–Ω–µ—á–Ω–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å) } $$ $$t - \text{–í—Ä–µ–º—è} (—Å) $$

$$\boxed{ S_{x} = V_{0_{x}}\cdot t + {{a_{x} \cdot t^2 \over 2}} \text { [–º]}}$$ $$S_{x} - \text{–ø–µ—Ä–µ–º–µ—â–µ–Ω–∏–µ(–º)} $$ $$V_{0_{x}} - \text{–Ω–∞—á–∞–ª—å–Ω–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å)}\\$$ $$a_{x} - \text{—É—Å–∫–æ—Ä–µ–Ω–∏–µ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å¬≤)} \\$$ $$t - \text{—Ä–∞—Å—Å–º–∞—Ç—Ä–∏–≤–∞–µ–º—ã–π –ø—Ä–æ–º–µ–∂—É—Ç–æ–∫ –≤—Ä–µ–º–µ–Ω–∏ (—Å) } \\$$

5) –ó–∞–≤–∏—Å–∏–º–æ—Å—Ç—å —Å–∫–æ—Ä–æ—Å—Ç–∏ –æ—Ç –≤—Ä–µ–º–µ–Ω–∏ –ø—Ä–∏ —Ä–∞–≤–Ω–æ—É—Å–∫–æ—Ä–µ–Ω–Ω–æ–º –¥–≤–∏–∂–µ–Ω–∏–∏

$$\boxed{ V_x(t) = V_{0_{x}} + {{a_{x} \cdot t}} \text { [–º/c]}}\\$$ $$V_{0_{x}} - \text{–Ω–∞—á–∞–ª—å–Ω–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è(–º/—Å)}$$ $$a_{x} - \text{—É—Å–∫–æ—Ä–µ–Ω–∏–µ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å¬≤)} $$ $$t \text{ - —Ä–∞—Å—Å–º–∞—Ç—Ä–∏–≤–∞–µ–º—ã–π –ø—Ä–æ–º–µ–∂—É—Ç–æ–∫ –≤—Ä–µ–º–µ–Ω–∏ (—Å) } $$ $$V_{x} \text{ - –∫–æ–Ω–µ—á–Ω–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å)} $$

–î–∏–Ω–∞–º–∏–∫–∞

6) –°–∏–ª–∞ —Ç—è–∂–µ—Å—Ç–∏

$$ \boxed{\vec{F} = {m} \cdot {\vec{g}} \text{ [H]}}$$ $$ \vec{F} - \text{—Å–∏–ª–∞ —Ç—è–∂–µ—Å—Ç–∏ (H)}¬†$$ $$ m - \text{ –º–∞—Å—Å–∞ —Ç–µ–ª–∞ (–∫–≥)}¬†$$ $$ {\vec{g}} - \text{—É—Å–∫–æ—Ä–µ–Ω–∏–µ —Å–≤–æ–±–æ–¥–Ω–æ–≥–æ –ø–∞–¥–µ–Ω–∏—è (–º/—Å¬≤)} $$

7) –°–∏–ª–∞ —É–ø—Ä—É–≥–æ—Å—Ç–∏

$$ \boxed{{\vec{F}_{—É–ø—Ä}} = {-k} \cdot {\Delta\vec{l}} \text{ [H]}}$$ $$ {\vec{F}_{—É–ø—Ä}} - \text{—Å–∏–ª–∞ —É–ø—Ä—É–≥–æ—Å—Ç–∏ (H)}¬†$$ $$ {k} - \text{–∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç –∂–µ—Å—Ç–∫–æ—Å—Ç–∏ —Ç–µ–ª–∞ (H/–º)} $$ $$ {\Delta\vec{l}} - \text{—É–¥–ª–∏–Ω–µ–Ω–∏–µ –ø—Ä—É–∂–∏–Ω—ã (–º)} $$

8) –°–∏–ª–∞ —Ç—Ä–µ–Ω–∏—è —Å–∫–æ–ª—å–∂–µ–Ω–∏—è

$$ \boxed{{{F}_{—Ç—Ä}} = {Œº} \cdot {N} \text{ [H]}}$$ $$ {{F}_{—Ç—Ä}} - \text{—Å–∏–ª–∞ —Ç—Ä–µ–Ω–∏—è (H)}¬†$$ $$ {G} - \text{–≥—Ä–∞–≤–∏—Ç–∞—Ü–∏–æ–Ω–Ω–∞—è –ø–æ—Å—Ç–æ—è–Ω–Ω–∞—è (H ¬∑ –º¬≤ / –∫–≥¬≤ )}¬†$$ $$ {Œº} - \text{–∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç —Ç—Ä–µ–Ω–∏—è} $$ $$ N - \text{—Å–∏–ª–∞ —Ä–µ–∞–∫—Ü–∏–∏ –æ–ø–æ—Ä—ã (H)} $$

9) –°–∏–ª–∞ –≤—Å–µ–º–∏—Ä–Ω–æ–≥–æ –≥—Ä–∞–≤–∏—Ç–∞—Ü–∏–æ–Ω–Ω–æ–≥–æ –ø—Ä–∏—Ç—è–∂–µ–Ω–∏—è

$$ \boxed{{F} = {{G} } \cdot{m_{1} \cdot m_{2} \over r^2 } \text{ [H]}}$$ $$ {F} - \text{—Å–∏–ª–∞ –≥—Ä–∞–≤–∏—Ç–∞—Ü–∏–æ–Ω–Ω–æ–≥–æ –≤–∞–∑–∏–º–æ–¥–µ–π—Å—Ç–≤–∏—è (H)}¬†$$ $$ m_{1} , m_{2} - \text{–º–∞—Å—Å–∞ –≤–∑–∞–∏–º–æ–¥–µ–π—Å—Ç–≤—É—é—â–∏—Ö —Ç–µ–ª (–∫–≥)} $$ $$ r - \text{ —Ä–∞—Å—Å—Ç–æ—è–Ω–∏–µ –º–µ–∂–¥—É —Ç–µ–ª–∞–º–∏ (–º)}¬†$$

10) –°–∏–ª–∞ –ê—Ä—Ö–∏–º–µ–¥–∞

$$ \boxed{{\vec{F}_{–∞—Ä—Ö}} = {-œÅ} \cdot {\vec{g}} \cdot {V_{–ø–æ–≥—Ä—É–∂}}\text{ [H]}}$$ $$ {\vec{F}_{–∞—Ä—Ö}} - \text{—Å–∏–ª–∞ –ê—Ä—Ö–∏–º–µ–¥–∞ (H)}¬†$$ $$ {œÅ} - \text{–ø–ª–æ—Ç–Ω–æ—Å—Ç—å –∂–∏–¥–∫–æ—Å—Ç–∏ –∏–ª–∏ –≥–∞–∑–∞ (–∫–≥/–º¬≥)} $$ $$ {\vec{g}} - \text{—É—Å–∫–æ—Ä–µ–Ω–∏–µ —Å–≤–æ–±–æ–¥–Ω–æ–≥–æ –ø–∞–¥–µ–Ω–∏—è (–º/—Å¬≤)} $$ $$ V_{–ø–æ–≥—Ä—É–∂} - \text{ –æ–±—ä–µ–º –ø–æ–≥—Ä—É–∂–µ–Ω–Ω–æ–π —á–∞—Å—Ç–∏ —Ç–µ–ª–∞ (–º¬≥)}¬†$$

11) –í—Ç–æ—Ä–æ–π –∑–∞–∫–æ–Ω –ù—å—é—Ç–æ–Ω–∞

$$ \boxed{\vec{a} = \frac {\vec F} m\text{ [–º/—Å¬≤]}}$$ $$ \vec{a} - \text{—É—Å–∫–æ—Ä–µ–Ω–∏–µ —Ç–µ–ª–∞ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å¬≤)} $$ $$ \vec {F} - \text{—Å—É–º–º–∞ –≤—Å–µ—Ö —Å–∏–ª –¥–µ–π—Å—Ç–≤—É—é—â–∏—Ö –Ω–∞ —Ç–µ–ª–æ, —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (H)} $$ $$m - \text{–º–∞—Å—Å–∞ —Ç–µ–ª–∞ (–∫–≥)}¬†$$

12) –¢—Ä–µ—Ç–∏–π –∑–∞–∫–æ–Ω –ù—å—é—Ç–æ–Ω–∞

$$ \boxed{\vec{F_{1_{2}}} = \vec{ -{F_{2_{1}}}}}$$ $$ \vec{F_{1_{2}}} - \text{—Å–∏–ª–∞, –¥–µ–π—Å—Ç–≤—É—é—â–∞—è –Ω–∞ –ø–µ—Ä–≤–æ–µ —Ç–µ–ª–æ —Å–æ —Å—Ç–æ—Ä–æ–Ω—ã –≤—Ç–æ—Ä–æ–≥–æ —Ç–µ–ª–∞ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (H)} $$ $$ \vec{ -F_{2_{1}}} - \text{—Å–∏–ª–∞, –¥–µ–π—Å—Ç–≤—É—é—â–∞—è –Ω–∞ –≤—Ç–æ—Ä–æ–µ —Ç–µ–ª–æ —Å–æ —Å—Ç–æ—Ä–æ–Ω—ã –ø–µ—Ä–≤–æ–≥–æ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (H)} $$


–¢—Ä–µ—Ç–∏–π –∑–∞–∫–æ–Ω –ù—å—é—Ç–æ–Ω–∞ - –ø—Ä–∏ –≤–∑–∞–∏–º–æ–¥–µ–π—Å—Ç–≤–∏–∏ –¥–≤—É—Ö —Ç–µ–ª –≤–æ–∑–Ω–∏–∫–∞–µ—Ç –ø–∞—Ä–∞ —Å–∏–ª, –∫–æ—Ç–æ—Ä—ã–µ

$$ \boxed{M = {F \cdot L} \text{ [–ù ¬∑ –º]}} $$ $$M - \text{–º–æ–º–µ–Ω—Ç —Å–∏–ª—ã (–ù ¬∑ –º)}$$ $$F - \text{—Å–∏–ª–∞ (–ù)} $$ $$L - \text{–ø–ª–µ—á–æ —Å–∏–ª—ã (–º)} $$

19) –ö–∏–Ω–µ—Ç–∏—á–µ—Å–∫–∞—è —ç–Ω–µ—Ä–≥–∏—è

$$\boxed{ {K} = {{m}\ \cdot v^2 \over 2 } \text{ [–î–∂]}}$$ $$K - \text{–∫–∏–Ω–µ—Ç–∏—á–µ—Å–∫–∞—è —ç–Ω–µ—Ä–≥–∏—è (–î–∂)}$$ $$m - \text{–º–∞—Å—Å–∞ —Ç–µ–ª–∞ (–∫–≥)}$$ $$v - \text{—Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ (–º/—Å)}$$

$$ \boxed{ùúë = {P_{t} \over P_{–Ω.–ø. t}} \ \cdot 100\% } $$ $$ùúë - \text{–æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–∞—è –≤–ª–∞–∂–Ω–æ—Å—Ç—å –≤–æ–∑–¥—É—Ö–∞}$$ $$P_{t} - \text{–¥–∞–≤–ª–µ–Ω–∏–µ –ø–∞—Ä–∞ –≤ –¥–∞–Ω–Ω—ã–π –º–æ–º–µ–Ω—Ç –≤—Ä–µ–º–µ–Ω–∏, –ø—Ä–∏ –¥–∞–Ω–Ω–æ–π —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä–µ (–ü–∞)}$$ $$P_{–Ω.–ø. t} - \text{–¥–∞–≤–ª–µ–Ω–∏–µ –Ω–∞—Å—ã—â–µ–Ω–Ω–æ–≥–æ –ø–∞—Ä–∞ –ø—Ä–∏ –¥–∞–Ω–Ω–æ–π —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä–µ (–ü–∞)}$$

37) –ê–¥–∏–∞–±–∞—Ç–∏—á–µ—Å–∫–∏–π –ø—Ä–æ—Ü–µ—Å—Å (–æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ+—Ñ–æ—Ä–º—É–ª–∞)

38) –†–∞–±–æ—Ç–∞ –∏–¥–µ–∞–ª—å–Ω–æ–≥–æ –≥–∞–∑–∞ (—Ñ–æ—Ä–º—É–ª–∞)
