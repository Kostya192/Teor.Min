–†–∞–≤–Ω–æ–º–µ—Ä–Ω–æ–µ –¥–≤–∏–∂–µ–Ω–∏–µ

1) –°—Ä–µ–¥–Ω—è—è —Å–∫–æ—Ä–æ—Å—Ç—å

$$\boxed{\vec{V} = \frac {\vec S} t \text{ [–º/—Å]}} $$ $$\vec{V} - \text{—Å—Ä–µ–¥–Ω—è—è —Å–∫–æ—Ä–æ—Å—Ç—å —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å)} $$ $$\vec S - \text{–ø–µ—Ä–µ–º–µ—â–µ–Ω–∏–µ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º)} $$ $$t - \text{–≤—Ä–µ–º—è –∑–∞ –∫–æ—Ç–æ—Ä–æ–µ —Å–¥–µ–ª–∞–Ω–æ –ø–µ—Ä–µ–º–µ—â–µ–Ω–∏–µ¬†(—Å)} $$

2) –°—Ä–µ–¥–Ω–µ–ø—É—Ç–µ–≤–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å

$$\boxed{{V} = \frac {L} t \text{ [–º/—Å]}}$$ $$V - \text{—Å—Ä–µ–¥–Ω–µ–ø—É—Ç–µ–≤–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å (–º/—Å)}$$ $$L - \text{–ø—É—Ç—å (–º)}$$ $$t¬†-¬†\text{–≤—Ä–µ–º—è¬†(—Å)}$$

3) –ó–∞–≤–∏—Å–∏–º–æ—Å—Ç—å –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç –æ—Ç –≤—Ä–µ–º–µ–Ω–∏ –ø—Ä–∏ —Ä–∞–≤–Ω–æ–º–µ—Ä–Ω–æ–º –¥–≤–∏–∂–µ–Ω–∏–∏

$$\boxed{X = X_{0} + V_x \cdot t \text{ [–º]}}$$ $$X_{0} - \text{–Ω–∞—á–∞–ª—å–Ω–∞—è –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞ —Ç–µ–ª–∞ (–º)}$$ $$X - \text{–∫–æ–Ω–µ—á–Ω–∞—è –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞ —Ç–µ–ª–∞ (–º)}$$ $$V_{x} \text{ - –∫–æ–Ω–µ—á–Ω–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å)} $$ $$t \text{ - –≤—Ä–µ–º—è (—Å) } $$

–†–∞–≤–Ω–æ—É—Å–∫–æ—Ä–µ–Ω–Ω–æ–µ –¥–≤–∏–∂–µ–Ω–∏–µ

4) –£—Å–∫–æ—Ä–µ–Ω–∏–µ

$$\boxed{\vec{a} = \frac {\vec V–∫- \vec V–Ω} t \text{ [–º/—Å¬≤]}}$$ $$\vec{a} - \text{—É—Å–∫–æ—Ä–µ–Ω–∏–µ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å¬≤)} $$ $$\vec V–∫ - \text{–∫–æ–Ω–µ—á–Ω–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å)} $$ $$\vec V–Ω - \text{–Ω–∞—á–∞–ª—å–Ω–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å)} $$ $$t¬†-\text{¬†–≤—Ä–µ–º—è(—Å)} $$

5) –ó–∞–≤–∏—Å–∏–º–æ—Å—Ç—å –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç –æ—Ç –≤—Ä–µ–º–µ–Ω–∏

$$\boxed{X(t) = X_{0} + V_{0_{x}}\cdot t + {{a_{x} \cdot t^2 \over 2}} \text { [–º]}}$$ $$X_{0} - \text{–Ω–∞—á–∞–ª—å–Ω–∞—è –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞ —Ç–µ–ª–∞ (–º)}$$ $$V_{0_{x}} - \text{–Ω–∞—á–∞–ª—å–Ω–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å)}$$ $$a_{x} - \text{—É—Å–∫–æ—Ä–µ–Ω–∏–µ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å¬≤)} $$ $$t \text{ - —Ä–∞—Å—Å–º–∞—Ç—Ä–∏–≤–∞–µ–º—ã–π –ø—Ä–æ–º–µ–∂—É—Ç–æ–∫ –≤—Ä–µ–º–µ–Ω–∏ (—Å) } $$ $$X \text{ - –∫–æ–Ω–µ—á–Ω–∞—è –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞ —Ç–µ–ª–∞ (–º)} $$

6) –ó–∞–≤–∏—Å–∏–º–æ—Å—Ç—å —Å–∫–æ—Ä–æ—Å—Ç–∏ –æ—Ç –≤—Ä–µ–º–µ–Ω–∏

$$\boxed{ V_x(t) = V_{0_{x}} + {{a_{x} \cdot t}} \text { [–º/c]}}$$ $$V_{0_{x}} - \text{–Ω–∞—á–∞–ª—å–Ω–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è(–º/—Å)}$$ $$a_{x} - \text{—É—Å–∫–æ—Ä–µ–Ω–∏–µ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å¬≤)} $$ $$t \text{ - —Ä–∞—Å—Å–º–∞—Ç—Ä–∏–≤–∞–µ–º—ã–π –ø—Ä–æ–º–µ–∂—É—Ç–æ–∫ –≤—Ä–µ–º–µ–Ω–∏ (—Å) } $$ $$V_{x} \text{ - –∫–æ–Ω–µ—á–Ω–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å)} $$

7) –§–æ—Ä–º—É–ª–∞ –ø–µ—Ä–µ–º–µ—â–µ–Ω–∏—è –±–µ–∑ –≤—Ä–µ–º–µ–Ω–∏

$$\boxed{ S_{x} = {V_{–ö_{x}}^2 - V_{0_{x}}^2 \over 2a_{x}} \text { [–º]}}$$ $$S_{x} - \text{–ø–µ—Ä–µ–º–µ—â–µ–Ω–∏–µ(–º)} $$ $$V_{0_{x}} - \text{–Ω–∞—á–∞–ª—å–Ω–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è(–º/—Å)} $$ $$V_{–ö_{x}} \text{ - –∫–æ–Ω–µ—á–Ω–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å)} $$ $$a_{x} - \text{—É—Å–∫–æ—Ä–µ–Ω–∏–µ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å¬≤)} $$

8) –§–æ—Ä–º—É–ª–∞ –ø–µ—Ä–µ–º–µ—â–µ–Ω–∏—è –±–µ–∑ —É—Å–∫–æ—Ä–µ–Ω–∏—è

$$ \boxed{S_{x} = {V_{–ö_{x}} + V_{0_{x}} \over 2} \cdot t \text{ [–º]}}$$ $$S_{x} - \text{–ø–µ—Ä–µ–º–µ—â–µ–Ω–∏–µ(–º)} $$ $$V_{0_{x}} - \text{–Ω–∞—á–∞–ª—å–Ω–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è(–º/—Å)} $$ $$V_{–ö_{x}} \text{ - –∫–æ–Ω–µ—á–Ω–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å) } $$ $$t - \text{–í—Ä–µ–º—è} (—Å) $$

–†–∞–≤–Ω–æ–º–µ—Ä–Ω–æ–µ –¥–≤–∏–∂–µ–Ω–∏–µ –ø–æ –æ–∫—Ä—É–∂–Ω–æ—Å—Ç–∏

9) –£–≥–ª–æ–≤–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å

$$ \boxed{\omega = {{ \Delta ùúë } \over \Delta t }\text{ [–†–∞–¥/c]}}$$ $$ \omega - \text{—É–≥–ª–æ–≤–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å (–†–∞–¥ /—Å)} $$ $$ \Delta ùúë \text{ - —É–≥–æ–ª –Ω–∞ –∫–æ—Ç–æ—Ä—ã–π –ø–æ–≤–æ—Ä–∞—á–∏–≤–∞–µ—Ç—Å—è —Ä–∞–¥–∏—É—Å –ø—Ä–æ–≤–µ–¥–µ–Ω–Ω—ã–π –∫ —Ç–æ—á–∫–µ –∑–∞ –≤—Ä–µ–º—è –¥–≤–∏–∂–µ–Ω–∏—è }\Delta t \text{(–†–∞–¥)} $$ $$ \Delta t - \text{–í—Ä–µ–º—è (—Å)}¬†$$

10) –£–≥–ª–æ–≤–æ–µ —É—Å–∫–æ—Ä–µ–Ω–∏–µ

$$\boxed{\varepsilon = {\Delta \omega \over \Delta t } \text{ [1/—Å¬≤]} }$$ $$\varepsilon \text{ - —É–≥–ª–æ–≤–æ–µ —É—Å–∫–æ—Ä–µ–Ω–∏–µ (1/—Å¬≤)} $$ $$\Delta \omega \text{ - –∏–∑–º–µ–Ω–µ–Ω–∏–µ —É–≥–ª–æ–≤–æ–π —Å–∫–æ—Ä–æ—Å—Ç–∏ (—Ä–∞–¥/—Å)} $$ $$\Delta t \text{ - –≤—Ä–µ–º—è (—Å)} $$

11) –ó–∞–≤–∏—Å–∏–º–æ—Å—Ç—å —É–≥–ª–∞ –æ—Ç –≤—Ä–µ–º–µ–Ω–∏

$$\boxed{ ùúë = ùúë_{0} + {\omega \cdot \Delta t } \text{ [—Ä–∞–¥]} }$$ $$ùúë \text{ - –∫–æ–Ω–µ—á–Ω–∞—è –≥—Ä–∞–¥—É—Å–Ω–∞—è –º–µ—Ä–∞ (—Ä–∞–¥)} $$ $$ùúë_{0} \text{ - –Ω–∞—á–∞–ª—å–Ω–∞—è –≥—Ä–∞–¥—É—Å–Ω–∞—è –º–µ—Ä–∞ (—Ä–∞–¥)} $$ $$ \omega - \text{—É–≥–ª–æ–≤–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å (–†–∞–¥ /—Å)} $$ $$\Delta t \text{ - –≤—Ä–µ–º—è (—Å)} $$

12) –ó–∞–≤–∏—Å–∏–º–æ—Å—Ç—å —É–≥–ª–æ–≤–æ–π —Å–∫–æ—Ä–æ—Å—Ç–∏ –æ—Ç –≤—Ä–µ–º–µ–Ω–∏

$$\boxed{\omega = \omega_{0} + {\varepsilon \cdot \Delta t } \text{ [—Ä–∞–¥]} } $$ $$ \omega - \text{—É–≥–ª–æ–≤–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å (–†–∞–¥ /—Å)} $$ $$ \omega_{0} - \text{–Ω–∞—á–∞–ª—å–Ω–∞—è —É–≥–ª–æ–≤–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å (–†–∞–¥ /—Å)} $$ $$\varepsilon \text{ - —É–≥–ª–æ–≤–æ–µ —É—Å–∫–æ—Ä–µ–Ω–∏–µ (1/—Å¬≤)} $$ $$\Delta t \text{ - –≤—Ä–µ–º—è (—Å)} $$

13) –°–≤—è–∑—å –º–µ–∂–¥—É –ª–∏–Ω–µ–π–Ω–æ–π –∏ —É–≥–ª–æ–≤–æ–π —Å–∫–æ—Ä–æ—Å—Ç—å—é

$$ \boxed{V = {\omega \cdot R }\text{ [–º/—Å]}}$$ $$ V - \text{—Å–∫–æ—Ä–æ—Å—Ç—å (–º/—Å)} $$ $$\omega - \text{—É–≥–ª–æ–≤–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å (—Ä–∞–¥/—Å)} $$ $$R - \text{–†–∞–¥–∏—É—Å –æ–∫—Ä—É–∂–Ω–æ—Å—Ç–∏ (–º)}¬†$$

14) –°–≤—è–∑—å —Ü–µ–Ω—Ç—Ä–æ—Å—Ç—Ä–µ–º–∏—Ç–µ–ª—å–Ω–æ–≥–æ —É—Å–∫–æ—Ä–µ–Ω–∏—è –∏ —É–≥–ª–æ–≤–æ–π —Å–∫–æ—Ä–æ—Å—Ç–∏

$$ \boxed{a_{—Ü} = {\omega^2 \cdot R }\text{ [–º/—Å¬≤]}}$$ $$ a_{—Ü} - \text{—Ü–µ–Ω—Ç—Ä–æ—Å—Ç—Ä–µ–º–∏—Ç–µ–ª—å–Ω–æ–µ —É—Å–∫–æ—Ä–µ–Ω–∏–µ (–º/—Å¬≤)} $$ $$\omega - \text{—É–≥–ª–æ–≤–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å (—Ä–∞–¥/—Å)} $$ $$R - \text{–†–∞–¥–∏—É—Å –æ–∫—Ä—É–∂–Ω–æ—Å—Ç–∏ (–º)}$$¬†

–î–∏–Ω–∞–º–∏–∫–∞

15) –°–∏–ª–∞ —Ç—è–∂–µ—Å—Ç–∏

$$ \boxed{\vec{F} = {m} \cdot {\vec{g}} \text{ [H]}}$$ $$ \vec{F} - \text{—Å–∏–ª–∞ —Ç—è–∂–µ—Å—Ç–∏ (H)}¬†$$ $$ {\vec{g}} - \text{—É—Å–∫–æ—Ä–µ–Ω–∏–µ —Å–≤–æ–±–æ–¥–Ω–æ–≥–æ –ø–∞–¥–µ–Ω–∏—è (–º/—Å¬≤)} $$ $$ m - \text{ –º–∞—Å—Å–∞ —Ç–µ–ª–∞ (–∫–≥)}¬†$$

16) –ó–∞–∫–æ–Ω –ì—É–∫–∞

$$ \boxed{{\vec{F}_{—É–ø—Ä}} = {-k} \cdot {\Delta\vec{l}} \text{ [H]}}$$ $$ {\vec{F}_{—É–ø—Ä}} - \text{—Å–∏–ª–∞ —É–ø—Ä—É–≥–æ—Å—Ç–∏ (H)}¬†$$ $$ {k} - \text{–∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç –∂–µ—Å—Ç–∫–æ—Å—Ç–∏ —Ç–µ–ª–∞ (H/–º)} $$ $$ {\Delta\vec{l}} - \text{—É–¥–ª–∏–Ω–µ–Ω–∏–µ –ø—Ä—É–∂–∏–Ω—ã (–º)} $$

17) –°–∏–ª–∞ —Ç—Ä–µ–Ω–∏—è —Å–∫–æ–ª—å–∂–µ–Ω–∏—è

$$ \boxed{{{F}_{—Ç—Ä}} = {Œº} \cdot {N} \text{ [H]}}$$ $$ {{F}_{—Ç—Ä}} - \text{—Å–∏–ª–∞ —Ç—Ä–µ–Ω–∏—è (H)}¬†$$ $$ {Œº} - \text{–∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç —Ç—Ä–µ–Ω–∏—è} $$ $$ N - \text{—Å–∏–ª–∞ —Ä–µ–∞–∫—Ü–∏–∏ –æ–ø–æ—Ä—ã (H)} $$

18) –°–∏–ª–∞ –≥—Ä–∞–≤–∏—Ç–∞—Ü–∏–æ–Ω–Ω–æ–≥–æ –≤–∑–∞–∏–º–æ–¥–µ–π—Å—Ç–≤–∏—è (+ —É—Å–ª–æ–≤–∏—è –ø—Ä–∏–º–µ–Ω–∏–º–æ—Å—Ç–∏)

$$ \boxed{{F} = {{G} } \cdot{m_{1} \cdot m_{2} \over r^2 } \text{ [H]}}$$ $$ {F} - \text{—Å–∏–ª–∞ –≥—Ä–∞–≤–∏—Ç–∞—Ü–∏–æ–Ω–Ω–æ–≥–æ –≤–∞–∑–∏–º–æ–¥–µ–π—Å—Ç–≤–∏—è (H)}¬†$$ $$ {G} - \text{–≥—Ä–∞–≤–∏—Ç–∞—Ü–∏–æ–Ω–Ω–∞—è –ø–æ—Å—Ç–æ—è–Ω–Ω–∞—è (H ¬∑ –º¬≤ / –∫–≥¬≤ )}¬†$$ $$ m_{1} , m_{2} - \text{–º–∞—Å—Å–∞ –≤–∑–∞–∏–º–æ–¥–µ–π—Å—Ç–≤—É—é—â–∏—Ö —Ç–µ–ª (–∫–≥)} $$ $$ r - \text{ —Ä–∞—Å—Å—Ç–æ—è–Ω–∏–µ –º–µ–∂–¥—É —Ç–µ–ª–∞–º–∏ (–º)}¬†$$

–ó–∞–∫–æ–Ω—ã —Å–æ—Ö—Ä–∞–Ω–µ–Ω–∏—è

19) –†–∞–±–æ—Ç–∞

$$\boxed{ {A} = {F \cdot \Delta X \cdot cos\ ùúë } \text{ [–î–∂]}}$$ $$A - \text{—Ä–∞–±–æ—Ç–∞ (–î–∂)}$$ $$F - \text{–ø–æ—Å—Ç–æ—è–Ω–Ω–∞—è —Å–∏–ª–∞, –¥–µ–π—Å—Ç–≤—É—é—â–∞—è –Ω–∞ —Ç–µ–ª–æ (–ù)}$$ $$\Delta X - \text{–∏–∑–º–µ–Ω–µ–Ω–∏–µ –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç —Ç–µ–ª–∞ (–º)}$$ $$cos\ ùúë - \text{–∫–æ—Å–∏–Ω—É—Å —É–≥–ª–∞ –º–µ–∂–¥—É –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏–µ–º —Å–∏–ª—ã –∏ –ø–µ—Ä–µ–º–µ—â–µ–Ω–∏–µ–º}$$

20) –ú–æ—â–Ω–æ—Å—Ç—å

$$\boxed{ {N} = {A \over t } \text{ [–í—Ç]}}$$ $$N - \text{–º–æ—â–Ω–æ—Å—Ç—å (–í—Ç)}$$ $$A - \text{—Ä–∞–±–æ—Ç–∞ (–î–∂)}$$ $$t - \text{–≤—Ä–µ–º—è (—Å)}$$

21) –¢–µ–æ—Ä–µ–º–∞ –æ –∫–∏–Ω–µ—Ç–∏—á–µ—Å–∫–æ–π —ç–Ω–µ—Ä–≥–∏–∏

$$\boxed{A = –ö_{–∫} - –ö_{–æ} }$$ $$A \text{ - —Ä–∞–±–æ—Ç–∞ (–î–∂)}$$ $$–ö_{–∫} \text{ - –∫–∏–Ω–µ—Ç–∏—á–µ—Å–∫–∞—è —ç–Ω–µ—Ä–≥–∏—è –≤ –Ω–∞—á–∞–ª–µ –¥–µ–π—Å—Ç–≤–∏—è (–î–∂)}$$ $$–ö_{–æ} \text{ - –∫–∏–Ω–µ—Ç–∏—á–µ—Å–∫–∞—è —ç–Ω–µ—Ä–≥–∏—è –≤ –∫–æ–Ω—Ü–µ –¥–µ–π—Å—Ç–≤–∏—è (–î–∂)}$$

22) –ö–∏–Ω–µ—Ç–∏—á–µ—Å–∫–∞—è —ç–Ω–µ—Ä–≥–∏—è

$$\boxed{ {K} = {{m}\ \cdot v^2 \over 2 } \text{ [–î–∂]}}$$ $$K - \text{–∫–∏–Ω–µ—Ç–∏—á–µ—Å–∫–∞—è —ç–Ω–µ—Ä–≥–∏—è (–î–∂)}$$ $$m - \text{–º–∞—Å—Å–∞ —Ç–µ–ª–∞ (–∫–≥)}$$ $$v - \text{—Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ (–º/—Å)}$$

23) –ü–æ—Ç–µ–Ω—Ü–∏–∞–ª—å–Ω–∞—è —ç–Ω–µ—Ä–≥–∏—è –≤–±–ª–∏–∑–∏ –ø–æ–≤–µ—Ä—Ö–Ω–æ—Å—Ç–∏ –ó–µ–º–ª–∏

$$\boxed{ {–ü} = {{m}\ \cdot g \cdot h } \text{ [–î–∂]}}$$ $$–ü - \text{–ø–æ—Ç–µ–Ω—Ü–∏–∞–ª—å–Ω–∞—è —ç–Ω–µ—Ä–≥–∏—è (–î–∂)}$$ $$m - \text{–º–∞—Å—Å–∞ —Ç–µ–ª–∞ (–∫–≥)}$$ $$g - \text{—É—Å–∫–æ—Ä–µ–Ω–∏–µ —Å–≤–æ–±–æ–¥–Ω–æ–≥–æ –ø–∞–¥–µ–Ω–∏—è (–º/—Å¬≤)}$$ $$h - \text{–≤—ã—Å–æ—Ç–∞ –Ω–∞–¥ –ø–æ–≤–µ—Ä—Ö–Ω–æ—Å—Ç—å—é –ó–µ–º–ª–∏ (–º)}$$

24) –ü–æ—Ç–µ–Ω—Ü–∏–∞–ª—å–Ω–∞—è —ç–Ω–µ—Ä–≥–∏—è –¥–µ—Ñ–æ—Ä–º–∏—Ä–æ–≤–∞–Ω–Ω–æ–π –ø—Ä—É–∂–∏–Ω—ã

$$\boxed{ {–ï} = {–ü + –ö} \text{ [–î–∂]}}$$ $$–ï - \text{–º–µ—Ö–∞–Ω–∏—á–µ—Å–∫–∞—è —ç–Ω–µ—Ä–≥–∏—è (–î–∂)}$$ $$–ü - \text{–ø–æ—Ç–µ–Ω—Ü–∏–∞–ª—å–Ω–∞—è —ç–Ω–µ—Ä–≥–∏—è (–î–∂)}$$ $$–ö - \text{–∫–∏–Ω–µ—Ç–∏—á–µ—Å–∫–∞—è —ç–Ω–µ—Ä–≥–∏—è (–î–∂)}$$

$$\boxed{ {–ï} = {–ï_{o} + –ê_{—Ç—Ä}+–ê_{–≤–Ω–µ—à–Ω}} \text{ [–î–∂]}}$$ $$–ï - \text{–º–µ—Ö–∞–Ω–∏—á–µ—Å–∫–∞—è —ç–Ω–µ—Ä–≥–∏—è (–î–∂)}$$ $$–ï_{o} - \text{–Ω–∞—á–∞–ª—å–Ω–∞—è –º–µ—Ö–∞–Ω–∏—á–µ—Å–∫–∞—è —ç–Ω–µ—Ä–≥–∏—è (–î–∂)}$$ $$–ê_{—Ç—Ä} - \text{—Ä–∞–±–æ—Ç–∞ –≤–Ω—É—Ç—Ä–µ–Ω–Ω–∏—Ö —Å–∏–ª —Ç—Ä–µ–Ω–∏—è (–î–∂)}$$ $$–ê_{–≤–Ω–µ—à–Ω} - \text{–≤–Ω–µ—à–Ω—è—è —Ä–∞–±–æ—Ç–∞ (–î–∂)}$$

27) –ò–º–ø—É–ª—å—Å

$$\boxed{\vec{P} = {m\ \cdot \vec{v} } \ \Big[{\text {–∫–≥} ¬∑ \text{–º} \over \text{—Å}}\Big]}$$ $$\vec{P} - \text{–∏–º–ø–ª—É–ª—å—Å}\ \Big({\text{–∫–≥} ¬∑ \text{–º} \over \text{—Å}}\Big)$$ $$m - \text{–º–∞—Å—Å–∞ —Ç–µ–ª–∞ (–∫–≥)}$$ $$\vec{v} - \text{—Å–∫–æ—Ä–æ—Å—Ç—å (–º/—Å)}$$

28) –í—Ç–æ—Ä–æ–π –∑–∞–∫–æ–Ω –ù—å—é—Ç–æ–Ω–∞ –≤ –∏–º–ø—É–ª—å—Å–Ω–æ–π —Ñ–æ—Ä–º–µ

$$\boxed{\Delta{\vec{P}} = {\vec{F}\ \cdot t } \ \Big[{\text{–∫–≥} ¬∑ \text{–º} \over \text{—Å}}\Big]}$$ $$\Delta{\vec{P}} - \text{–∏–º–ø–ª—É–ª—å—Å}\ \Big({\text{–∫–≥} ¬∑ \text{–º} \over \text{—Å}}\Big)$$ $$\vec{F} - \text{—Å–∏–ª–∞ (–ù)}$$ $$t - \text{–≤—Ä–µ–º—è (—Å)}$$

29) –ú–æ–º–µ–Ω—Ç —Å–∏–ª—ã

30) –î–∞–≤–ª–µ–Ω–∏–µ

$$ \boxed{P = {F_{‚üÇ} \over S} \text{ [–ü–∞]}} $$ $$ P - \text{–¥–∞–≤–ª–µ–Ω–∏–µ (–ü–∞)}$$ $$F_{‚üÇ} - \text{—Å–∏–ª–∞, –¥–µ–π—Å—Ç–≤—É—é—â–∞—è –ø–µ—Ä–ø–µ–Ω–¥–∏–∫—É–ª—è—Ä–Ω–æ –∫ –ø–æ–≤–µ—Ä—Ö–Ω–æ—Å—Ç–∏ (–ù)} $$ $$S - \text{–ø–ª–æ—â–∞–¥—å –ø–æ–≤–µ—Ä—Ö–Ω–æ—Å—Ç–∏ (–º¬≤)} $$

$$ \boxed{{\vec{F}_{–∞—Ä—Ö}} = {-œÅ} \cdot {\vec{g}} \cdot {V_{–ø–æ–≥—Ä—É–∂}}\text{ [H]}}$$ $$ {\vec{F}_{–∞—Ä—Ö}} - \text{—Å–∏–ª–∞ –ê—Ä—Ö–∏–º–µ–¥–∞ (H)}¬†$$ $$ {œÅ} - \text{–ø–ª–æ—Ç–Ω–æ—Å—Ç—å –∂–∏–¥–∫–æ—Å—Ç–∏ –∏–ª–∏ –≥–∞–∑–∞ (–∫–≥/–º¬≥)} $$ $$ {\vec{g}} - \text{—É—Å–∫–æ—Ä–µ–Ω–∏–µ —Å–≤–æ–±–æ–¥–Ω–æ–≥–æ –ø–∞–¥–µ–Ω–∏—è (–º/—Å¬≤)} $$ $$ V_{–ø–æ–≥—Ä—É–∂} - \text{ –æ–±—ä–µ–º –ø–æ–≥—Ä—É–∂–µ–Ω–Ω–æ–π —á–∞—Å—Ç–∏ —Ç–µ–ª–∞ (–º¬≥)}¬†$$

–ö–æ–ª–µ–±–∞–Ω–∏—è

33) –ü–µ—Ä–∏–æ–¥ –∫–æ–ª–µ–±–∞–Ω–∏–π –º–∞—Ç–µ–º–∞—Ç–∏—á–µ—Å–∫–æ–≥–æ –º–∞—è—Ç–Ω–∏–∫–∞

$$ \boxed{T = 2 \cdot œÄ \cdot{\sqrt{L \over g}}\text{ [—Å]}} $$ $$T - \text{–ø–µ—Ä–∏–æ–¥ (—Å)} $$ $$L - \text{–¥–ª–∏–Ω–∞ –Ω–∏—Ç–∏ (–º)} $$ $$g- \text{—É—Å–∫–æ—Ä–µ–Ω–∏–µ —Å–≤–æ–±–æ–¥–Ω–æ–≥–æ –ø–∞–¥–µ–Ω–∏—è (–º/—Å¬≤)} $$

34) –ü–µ—Ä–∏–æ–¥ –∫–æ–ª–µ–±–∞–Ω–∏—è –ø—Ä—É–∂–∏–Ω–Ω–æ–≥–æ –º–∞—è—Ç–Ω–∏–∫–∞

$$–ü—Ä–æ–¥–æ–ª—å–Ω—ã–º–∏ \ –≤–æ–ª–Ω–∞–º–∏ \text{ - –Ω–∞–∑—ã–≤–∞—é—Ç –≤–æ–ª–Ω—ã, –≤ –∫–æ—Ç–æ—Ä—ã—Ö –∫–æ–ª–µ–±–∞–Ω–∏—è —á–∞—Å—Ç–∏—Ü –ø—Ä–æ–∏—Å—Ö–æ–¥—è—Ç –≤–¥–æ–ª—å –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è —Ä–∞—Å–ø—Ä–æ—Å—Ç—Ä–∞–Ω–µ–Ω–∏—è –≤–æ–ª–Ω—ã} $$
