$$\vec{V} = \frac {\vec S} t \text{ [–º/—Å]} $$ $$\vec{V} - \text{—Å—Ä–µ–¥–Ω—è—è —Å–∫–æ—Ä–æ—Å—Ç—å —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å)} $$ $$\vec S - \text{–ø–µ—Ä–µ–º–µ—â–µ–Ω–∏–µ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º)} $$ $$t - \text{–≤—Ä–µ–º—è –∑–∞ –∫–æ—Ç–æ—Ä–æ–µ —Å–¥–µ–ª–∞–Ω–æ –ø–µ—Ä–µ–º–µ—â–µ–Ω–∏–µ¬†(—Å)} $$

2) –°—Ä–µ–¥–Ω–µ –ø—É—Ç–µ–≤–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å

$${V} = \frac {L} t \text{ [–º/—Å]}$$ $$V - \text{—Å—Ä–µ–¥–Ω–µ –ø—É—Ç–µ–≤–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å (–º/—Å)}$$ $$L - \text{–ø—É—Ç—å (–º)}$$ $$t¬†-¬†\text{–≤—Ä–µ–º—è¬†(—Å)}$$

–†–∞–≤–Ω–æ—É—Å–∫–æ—Ä–µ–Ω–Ω–æ–µ –¥–≤–∏–∂–µ–Ω–∏–µ

5) –ó–∞–≤–∏—Å–∏–º–æ—Å—Ç—å –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç –æ—Ç –≤—Ä–µ–º–µ–Ω–∏

$$ X(t) = X_{0} + V_{0_{x}} + {{a_{x} \cdot t^2 \over 2}} \text { [–º]}$$ $$X_{0} - \text{–Ω–∞—á–∞–ª—å–Ω–∞—è –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞ —Ç–µ–ª–∞ (–º)}\\$$ $$V_{0_{x}} - \text{–Ω–∞—á–∞–ª—å–Ω–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å)}\\$$ $$a_{x} - \text{—É—Å–∫–æ—Ä–µ–Ω–∏–µ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å¬≤)} \\$$ $$t \text{ - —Ä–∞—Å—Å–º–∞—Ç—Ä–∏–≤–∞–µ–º—ã–π –ø—Ä–æ–º–µ–∂—É—Ç–æ–∫ –≤—Ä–µ–º–µ–Ω–∏ (—Å) } \\$$ $$X \text{ - –∫–æ–Ω–µ—á–Ω–∞—è –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞ —Ç–µ–ª–∞ (–º)} $$

6) –ó–∞–≤–∏—Å–∏–º–æ—Å—Ç—å —Å–∫–æ—Ä–æ—Å—Ç–∏ –æ—Ç –≤—Ä–µ–º–µ–Ω–∏

$$ V_x(t) = V_{0_{x}} + {{a_{x} \cdot t}} \text { [–º/c]}\\$$ $$V_{0_{x}} - \text{–Ω–∞—á–∞–ª—å–Ω–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è(–º/—Å)}$$ $$a_{x} - \text{—É—Å–∫–æ—Ä–µ–Ω–∏–µ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å¬≤)} $$ $$t \text{ - —Ä–∞—Å—Å–º–∞—Ç—Ä–∏–≤–∞–µ–º—ã–π –ø—Ä–æ–º–µ–∂—É—Ç–æ–∫ –≤—Ä–µ–º–µ–Ω–∏ (—Å) } $$ $$V_{x} \text{ - –∫–æ–Ω–µ—á–Ω–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å)} $$

7) –§–æ—Ä–º—É–ª–∞ –ø–µ—Ä–µ–º–µ—â–µ–Ω–∏—è –±–µ–∑ –≤—Ä–µ–º–µ–Ω–∏

$$ S_{x} = {V_{–ö_{x}} + V_{0_{x}} \over 2_{x}} \cdot t \text{ [–º/c]}$$ $$S_{x} - \text{–ø–µ—Ä–µ–º–µ—â–µ–Ω–∏–µ(–º)} $$ $$V_{0_{x}} - \text{–Ω–∞—á–∞–ª—å–Ω–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è(–º/—Å)} $$ $$V_{–ö_{x}} \text{ - –∫–æ–Ω–µ—á–Ω–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å) } $$ $$t - \text{–í—Ä–µ–º—è} (—Å) $$

–†–∞–≤–Ω–æ–º–µ—Ä–Ω–æ–µ –¥–≤–∏–∂–µ–Ω–∏–µ –ø–æ –æ–∫—Ä—É–∂–Ω–æ—Å—Ç–∏

$$ \omega = {{ \Delta ùúë } \over \Delta t }\text{ [–†–∞–¥/c]}$$ $$ \omega - \text{—É–≥–ª–æ–≤–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å (–†–∞–¥ /—Å)} $$ $$ \Delta ùúë \text{ - —É–≥–æ–ª –Ω–∞ –∫–æ—Ç–æ—Ä—ã–π –ø–æ–≤–æ—Ä–∞—á–∏–≤–∞–µ—Ç—Å—è —Ä–∞–¥–∏—É—Å –ø—Ä–æ–≤–µ–¥–µ–Ω–Ω—ã–π –∫ —Ç–æ—á–∫–µ –∑–∞ –≤—Ä–µ–º—è –¥–≤–∏–∂–µ–Ω–∏—è }\Delta t \text{(–†–∞–¥)} $$ $$ \Delta t - \text{–í—Ä–µ–º—è (—Å)}¬†$$
