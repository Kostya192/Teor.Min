–§–æ—Ä–º—É–ª—ã

1) –ó–∞–≤–∏—Å–∏–º–æ—Å—Ç—å –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç –æ—Ç –≤—Ä–µ–º–µ–Ω–∏ –ø—Ä–∏ —Ä–∞–≤–Ω–æ—É—Å–∫–æ—Ä–µ–Ω–Ω–æ–º –¥–≤–∏–∂–µ–Ω–∏–∏

$$\boxed{X(t) = X_{0} + V_{0_{x}}\cdot t + {{a_{x} \cdot t^2 \over 2}} \text { [–º]}}$$ $$X \text{ - –∫–æ–Ω–µ—á–Ω–∞—è –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞ —Ç–µ–ª–∞ (–º)} $$ $$X_{0} - \text{–Ω–∞—á–∞–ª—å–Ω–∞—è –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞ —Ç–µ–ª–∞ (–º)}\\$$ $$V_{0_{x}} - \text{–Ω–∞—á–∞–ª—å–Ω–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å)}$$ $$t \text{ - —Ä–∞—Å—Å–º–∞—Ç—Ä–∏–≤–∞–µ–º—ã–π –ø—Ä–æ–º–µ–∂—É—Ç–æ–∫ –≤—Ä–µ–º–µ–Ω–∏ (—Å) } $$ $$a_{x} - \text{—É—Å–∫–æ—Ä–µ–Ω–∏–µ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å¬≤)} $$

2) –ó–∞–≤–∏—Å–∏–º–æ—Å—Ç—å —Å–∫–æ—Ä–æ—Å—Ç–∏ –æ—Ç –≤—Ä–µ–º–µ–Ω–∏ –ø—Ä–∏ —Ä–∞–≤–Ω–æ—É—Å–∫–æ—Ä–µ–Ω–Ω–æ–º –¥–≤–∏–∂–µ–Ω–∏–∏

$$\boxed{ V_x(t) = V_{0_{x}} + {{a_{x} \cdot t}} \text { [–º/c]}}$$ $$V_{0_{x}} - \text{–Ω–∞—á–∞–ª—å–Ω–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è(–º/—Å)}$$ $$a_{x} - \text{—É—Å–∫–æ—Ä–µ–Ω–∏–µ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å¬≤)} $$ $$t \text{ - —Ä–∞—Å—Å–º–∞—Ç—Ä–∏–≤–∞–µ–º—ã–π –ø—Ä–æ–º–µ–∂—É—Ç–æ–∫ –≤—Ä–µ–º–µ–Ω–∏ (—Å) } $$ $$V_{x} \text{ - –∫–æ–Ω–µ—á–Ω–∞—è —Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ —Å —É—á–µ—Ç–æ–º –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è (–º/—Å)} $$

3) –°–∏–ª–∞ —Ç—è–∂–µ—Å—Ç–∏

$$ \boxed{\vec{F} = {m} \cdot {\vec{g}} \text{ [H]}}$$ $$ \vec{F} - \text{—Å–∏–ª–∞ —Ç—è–∂–µ—Å—Ç–∏ (H)}¬†$$ $$ {\vec{g}} - \text{—É—Å–∫–æ—Ä–µ–Ω–∏–µ —Å–≤–æ–±–æ–¥–Ω–æ–≥–æ –ø–∞–¥–µ–Ω–∏—è (–º/—Å¬≤)} $$ $$ m - \text{ –º–∞—Å—Å–∞ —Ç–µ–ª–∞ (–∫–≥)}¬†$$

4) –°–∏–ª–∞ —É–ø—Ä—É–≥–æ—Å—Ç–∏

$$ \boxed{{\vec{F}_{—É–ø—Ä}} = {-k} \cdot {\Delta\vec{l}} \text{ [H]}}$$ $$ {\vec{F}_{—É–ø—Ä}} - \text{—Å–∏–ª–∞ —É–ø—Ä—É–≥–æ—Å—Ç–∏ (H)}¬†$$ $$ {k} - \text{–∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç –∂–µ—Å—Ç–∫–æ—Å—Ç–∏ —Ç–µ–ª–∞ (H/–º)} $$ $$ {\Delta\vec{l}} - \text{—É–¥–ª–∏–Ω–µ–Ω–∏–µ –ø—Ä—É–∂–∏–Ω—ã (–º)} $$

5) –°–∏–ª–∞ —Ç—Ä–µ–Ω–∏—è —Å–∫–æ–ª—å–∂–µ–Ω–∏—è

$$ \boxed{{{F}_{—Ç—Ä}} = {Œº} \cdot {N} \text{ [H]}}$$ $$ {{F}_{—É–ø—Ä}} - \text{—Å–∏–ª–∞ —É–ø—Ä—É–≥–æ—Å—Ç–∏ (H)}¬†$$ $$ {Œº} - \text{–∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç —Ç—Ä–µ–Ω–∏—è} $$ $$ N - \text{—Å–∏–ª–∞ —Ä–µ–∞–∫—Ü–∏–∏ –æ–ø–æ—Ä—ã (H)} $$

6) –°–∏–ª–∞ –≤—Å–µ–º–∏—Ä–Ω–æ–≥–æ –≥—Ä–∞–≤–∏—Ç–∞—Ü–∏–æ–Ω–Ω–æ–≥–æ –ø—Ä–∏—Ç—è–∂–µ–Ω–∏—è

$$ \boxed{{F} = {{G} } \cdot{m_{1} \cdot m_{2} \over r^2 } \text{ [H]}}$$ $$ {F} - \text{—Å–∏–ª–∞ –≥—Ä–∞–≤–∏—Ç–∞—Ü–∏–æ–Ω–Ω–æ–≥–æ –≤–∞–∑–∏–º–æ–¥–µ–π—Å—Ç–≤–∏—è (H)}¬†$$ $$ m_{1} , m_{2} - \text{–º–∞—Å—Å–∞ –≤–∑–∞–∏–º–æ–¥–µ–π—Å—Ç–≤—É—é—â–∏—Ö —Ç–µ–ª (–∫–≥)} $$ $$ r - \text{ —Ä–∞—Å—Å—Ç–æ—è–Ω–∏–µ –º–µ–∂–¥—É —Ç–µ–ª–∞–º–∏ (–º)}¬†$$

7) –ú–æ–º–µ–Ω—Ç —Å–∏–ª—ã

$$ \boxed{M = {F \cdot L} \text{ [–ù ¬∑ –º]}} $$ $$M - \text{–º–æ–º–µ–Ω—Ç —Å–∏–ª—ã (–ù ¬∑ –º)}$$ $$F - \text{—Å–∏–ª–∞ (–ù)} $$ $$L - \text{–ø–ª–µ—á–æ —Å–∏–ª—ã (–º)} $$

8) –£—Å–ª–æ–≤–∏—è —Ä–∞–≤–Ω–æ–≤–µ—Å–∏—è –ø—Ä–æ—Ç—è–∂—ë–Ω–Ω–æ–≥–æ —Ç–≤—ë—Ä–¥–æ–≥–æ —Ç–µ–ª–∞ (—Ñ–æ—Ä–º—É–ª–∏—Ä–æ–≤–∫–∞)

$$\text{1) –í–µ–∫—Ç–æ—Ä–Ω–∞—è —Å—É–º–º–∞ –≤—Å–µ—Ö —Å–∏–ª, –¥–µ–π—Å—Ç–≤—É—é—â–∏—Ö –Ω–∞ —Ç–µ–ª–æ —Ä–∞–≤–Ω–æ 0 . } $$ $$\text{2) –°—É–º–º–∞ –≤—Å–µ—Ö –º–æ–º–µ–Ω—Ç–æ–≤, –∑–∞–∫—Ä—É—á–∏–≤–∞—é—â–∏—Ö —Ä—ã—á–∞–≥ –ø–æ —á–∞—Å–æ–≤–æ–π —Å—Ç—Ä–µ–ª–∫–µ, —Ä–∞–≤–Ω–æ —Å—É–º–º–µ –≤—Å–µ—Ö –º–æ–º–µ–Ω—Ç–æ–≤, –∑–∞–∫—Ä—É—á–∏–≤–∞—é—â–∏—Ö —Ä—ã—á–∞–≥ –ø—Ä–æ—Ç–∏–≤ —á–∞—Å–æ–≤–æ–π —Å—Ç—Ä–µ–ª–∫–µ.} $$

9) –ú–µ—Ö–∞–Ω–∏—á–µ—Å–∫–∞—è —Ä–∞–±–æ—Ç–∞

$$\boxed{ {A} = {F \cdot \Delta X \cdot cos\ ùúë } \text{ [–î–∂]}}$$ $$A - \text{–ú–µ—Ö–∞–Ω–∏—á–µ—Å–∫–∞—è —Ä–∞–±–æ—Ç–∞ (–î–∂)}$$ $$F - \text{–ø–æ—Å—Ç–æ—è–Ω–Ω–∞—è —Å–∏–ª–∞, –¥–µ–π—Å—Ç–≤—É—é—â–∞—è –Ω–∞ —Ç–µ–ª–æ (–ù)}$$ $$\Delta X - \text{–∏–∑–º–µ–Ω–µ–Ω–∏–µ –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç —Ç–µ–ª–∞ (–º)}$$ $$cos\ ùúë - \text{–∫–æ—Å–∏–Ω—É—Å —É–≥–ª–∞ –º–µ–∂–¥—É –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏–µ–º —Å–∏–ª—ã –∏ –ø–µ—Ä–µ–º–µ—â–µ–Ω–∏–µ–º}$$

10) –ú–æ—â–Ω–æ—Å—Ç—å

$$\boxed{ {N} = {A \over t } \text{ [–í—Ç]}}$$ $$N - \text{–º–æ—â–Ω–æ—Å—Ç—å (–í—Ç)}$$ $$A - \text{—Ä–∞–±–æ—Ç–∞ (–î–∂)}$$ $$t - \text{–≤—Ä–µ–º—è (—Å)}$$

11) –ö–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç –ø–æ–ª–µ–∑–Ω–æ–≥–æ –¥–µ–π—Å—Ç–≤–∏—è

$$ \boxed{Œ∑ = {{A_{–ø–æ–ª–µ–∑} \over A_{–∑–∞—Ç—Ä}}} \ \cdot 100\% } $$ $$Œ∑ - \text{–ö.–ü.–î. (%)}$$ $$A_{–ø–æ–ª–µ–∑} - \text{—Ä–∞–±–æ—Ç–∞, –ø—Ä–æ–∏–∑–≤–æ–¥–∏–º–∞—è –º–µ—Ö–∞–Ω–∏–∑–º–æ–º –Ω–∞–¥ –ø–µ—Ä–µ–º–µ—â–∞–µ–º—ã–º —Ç–µ–ª–æ–º (–î–∂)}$$ $$A_{–∑–∞—Ç—Ä} - \text{—Ä–∞–±–æ—Ç–∞, –ø—Ä–æ–∏–∑–≤–æ–¥–∏–º–∞—è –Ω–∞–¥ –º–µ—Ö–∞–Ω–∏–∑–º–æ–º –¥–ª—è –ø—Ä–∏–≤–µ–¥–µ–Ω–∏—è –µ–≥–æ –≤ –¥–≤–∏–∂–µ–Ω–∏–µ (–î–∂)}$$

12) –ö–∏–Ω–µ—Ç–∏—á–µ—Å–∫–∞—è —ç–Ω–µ—Ä–≥–∏—è

$$\boxed{ {K} = {{m}\ \cdot v^2 \over 2 } \text{ [–î–∂]}}$$ $$K - \text{–∫–∏–Ω–µ—Ç–∏—á–µ—Å–∫–∞—è —ç–Ω–µ—Ä–≥–∏—è (–î–∂)}$$ $$m - \text{–º–∞—Å—Å–∞ —Ç–µ–ª–∞ (–∫–≥)}$$ $$v - \text{—Å–∫–æ—Ä–æ—Å—Ç—å —Ç–µ–ª–∞ (–º/—Å)}$$

13) –ü–æ—Ç–µ–Ω—Ü–∏–∞–ª—å–Ω–∞—è —ç–Ω–µ—Ä–≥–∏—è —Ç–µ–ª–∞ –≤–±–ª–∏–∑–∏ –ø–æ–≤–µ—Ä—Ö–Ω–æ—Å—Ç–∏ –ó–µ–º–ª–∏

$$\boxed{ {–ü} = {{m}\ \cdot g \cdot h } \text{ [–î–∂]}}$$ $$–ü - \text{–ø–æ—Ç–µ–Ω—Ü–∏–∞–ª—å–Ω–∞—è —ç–Ω–µ—Ä–≥–∏—è (–î–∂)}$$ $$m - \text{–º–∞—Å—Å–∞ —Ç–µ–ª–∞ (–∫–≥)}$$ $$g - \text{—É—Å–∫–æ—Ä–µ–Ω–∏–µ —Å–≤–æ–±–æ–¥–Ω–æ–≥–æ –ø–∞–¥–µ–Ω–∏—è (–º/—Å¬≤)}$$ $$h - \text{–≤—ã—Å–æ—Ç–∞ –Ω–∞–¥ –ø–æ–≤–µ—Ä—Ö–Ω–æ—Å—Ç—å—é –ó–µ–º–ª–∏ (–º)}$$

14) –ü–æ—Ç–µ–Ω—Ü–∏–∞–ª—å–Ω–∞—è —ç–Ω–µ—Ä–≥–∏—è —Å–∂–∞—Ç–æ–π –ø—Ä—É–∂–∏–Ω—ã

$$ \boxed{P = {F_{‚üÇ} \over S} \text{ [–ü–∞]}} $$ $$ P - \text{–¥–∞–≤–ª–µ–Ω–∏–µ (–ü–∞)}$$ $$F_{‚üÇ} - \text{—Å–∏–ª–∞, –¥–µ–π—Å—Ç–≤—É—é—â–∞—è –ø–µ—Ä–ø–µ–Ω–¥–∏–∫—É–ª—è—Ä–Ω–æ –∫ –ø–æ–≤–µ—Ä—Ö–Ω–æ—Å—Ç–∏ (–ù)} $$ $$S - \text{–ø–ª–æ—â–∞–¥—å –ø–æ–≤–µ—Ä—Ö–Ω–æ—Å—Ç–∏ (–º¬≤)} $$

17) –°–∏–ª–∞ –ê—Ä—Ö–∏–º–µ–¥–∞

$$ \boxed{{\vec{F}_{–∞—Ä—Ö}} = {-œÅ} \cdot {\vec{g}} \cdot {V_{–ø–æ–≥—Ä—É–∂}}\text{ [H]}}$$ $$ {\vec{F}_{–∞—Ä—Ö}} - \text{—Å–∏–ª–∞ –ê—Ä—Ö–∏–º–µ–¥–∞ (H)}¬†$$ $$ {œÅ} - \text{–ø–ª–æ—Ç–Ω–æ—Å—Ç—å –∂–∏–¥–∫–æ—Å—Ç–∏ –∏–ª–∏ –≥–∞–∑–∞ (–∫–≥/–º¬≥)} $$ $$ {\vec{g}} - \text{—É—Å–∫–æ—Ä–µ–Ω–∏–µ —Å–≤–æ–±–æ–¥–Ω–æ–≥–æ –ø–∞–¥–µ–Ω–∏—è (–º/—Å¬≤)} $$ $$ V_{–ø–æ–≥—Ä—É–∂} - \text{ –æ–±—ä–µ–º –ø–æ–≥—Ä—É–∂–µ–Ω–Ω–æ–π —á–∞—Å—Ç–∏ —Ç–µ–ª–∞ (–º¬≥)}¬†$$

18) –í–Ω—É—Ç—Ä–µ–Ω–Ω—è—è —ç–Ω–µ—Ä–≥–∏—è (–æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ)

$$–í–Ω—É—Ç—Ä–µ–Ω–Ω—è—è \ —ç–Ω–µ—Ä–≥–∏–∏ \text{ - —Å—É–º–º–∞ –∫–∏–Ω–µ—Ç–∏—á–µ—Å–∫–∏—Ö —ç–Ω–µ—Ä–≥–∏–π —Ö–∞–æ—Ç–∏—á–µ—Å–∫–æ–≥–æ –¥–≤–∏–∂–µ–Ω–∏—è –º–æ–ª–µ–∫—É–ª –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ —Ü–µ–Ω—Ç—Ä–∞ –º–∞—Å—Å –∏ –ø–æ—Ç–µ–Ω—Ü–∏–∞–ª—å–Ω—ã—Ö —ç–Ω–µ—Ä–≥–∏–π –≤–∑–∞–∏–º–æ–¥–µ–π—Å—Ç–≤–∏—è –º–æ–ª–µ–∫—É–ª –¥—Ä—É–≥ —Å –¥—Ä—É–≥–æ–º, –Ω–æ –Ω–µ —Å –º–æ–ª–µ–∫—É–ª–∞–º–∏ –¥—Ä—É–≥–æ–≥–æ —Ç–µ–ª–∞} $$

19) –¢–µ–ø–ª–æ—ë–º–∫–æ—Å—Ç—å —Ç–µ–ª–∞ (–æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ + —Ñ–æ—Ä–º—É–ª–∞)

$$ \boxed{C = {Q \over \Delta T} \text{ [–î–∂/–ö]}}$$ $$C - \text{–¢–µ–ø–ª–æ—ë–º–∫–æ—Å—Ç—å –≤–µ—â–µ—Å—Ç–≤–∞ (–î–∂/–ö)}$$ $$Q - \text{–∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Ç–µ–ø–ª–æ—Ç—ã (–î–∂)}$$ $$\Delta T - \text{—Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä–∞ (–ö)}$$

20) –£–¥–µ–ª—å–Ω–∞—è —Ç–µ–ø–ª–æ—ë–º–∫–æ—Å—Ç—å –≤–µ—â–µ—Å—Ç–≤–∞ (–æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ + —Ñ–æ—Ä–º—É–ª–∞)

-

$$ \boxed{c = {{Q }\over m \cdot \Delta T} \text{ [–î–∂]}} $$ $$c - \text{—É–¥–µ–ª—å–Ω–∞—è —Ç–µ–ø–ª–æ–µ–º–∫–æ—Å—Ç—å —Ç–µ–ª–∞} \ \Big({\text{–î–∂} \over {{\text{–∫–≥} \cdot \text{–ö}}}}\Big)$$ $$Q - \text{–∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Ç–µ–ø–ª–æ—Ç—ã (–î–∂)} $$ $$m - \text{–º–∞—Å—Å–∞ (–∫–≥)}$$ $$\Delta T - \text{–∏–∑–º–µ–Ω–µ–Ω–∏–µ —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä—ã (–ö)} $$

21) –£–¥–µ–ª—å–Ω–∞—è —Ç–µ–ø–ª–æ—Ç–∞ —Å–≥–æ—Ä–∞–Ω–∏—è —Ç–æ–ø–ª–∏–≤–∞ (–æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ + —Ñ–æ—Ä–º—É–ª–∞)
